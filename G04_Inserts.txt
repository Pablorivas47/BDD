create or replace function fnc_comprobante() RETURNS void AS $$
	DECLARE
		v_compr RECORD;
		v_turno_fecha timestamp;
		v_hour integer;
		v_total numeric(18,5);
    BEGIN
    	FOR v_compr IN SELECT * FROM comprobante LOOP

    		select desde
    		into v_turno_fecha
    		from turno
    		where id_turno = v_compr.id_turno;

    		-- Recibo
    		if (v_compr.id_tcomp = 3) then
	    		v_hour = floor(random()*8);
	    		update comprobante set fecha = v_turno_fecha + (''||v_hour||' hour')::interval,
	    			fecha_vencimiento = null,
	    			estado = 'PAGADO'
	    			where id_comp = v_compr.id_comp and
	    				id_tcomp = v_compr.id_tcomp;
	    	end if;

    		-- Factura
    		if (v_compr.id_tcomp = 1) then

    			select coalesce(sum(importe), 0)
    			into v_total
    			from lineacomprobante
    			where id_comp = v_compr.id_comp and
	    				id_tcomp = v_compr.id_tcomp;

				update comprobante set fecha_vencimiento = fecha + ' 10 day'::interval,
					importe = v_total,
					id_turno = null
    			where id_comp = v_compr.id_comp and
    				id_tcomp = v_compr.id_tcomp;
    		end if;

    		-- Remito
    		if (v_compr.id_tcomp = 2) then
				update comprobante set
					fecha_vencimiento = null,
					id_turno = null,
					importe = 0
    			where id_comp = v_compr.id_comp and
    				id_tcomp = v_compr.id_tcomp;
    		end if;

	    END LOOP;
    END;
$$ LANGUAGE plpgsql;

select fnc_comprobante();

update turno t
	set dinero_fin = dinero_inicio + (select coalesce(sum(importe),0) from comprobante where id_turno = t.id_turno);

	set saldo = (select coalesce(sum(importe * (-1)),0) from comprobante where id_cliente=c.id_cliente and id_tcomp = 1) +
				(select coalesce(sum(importe),0) from comprobante where id_cliente=c.id_cliente and id_tcomp = 3);

CREATE OR REPLACE FUNCTION generate_random_mac() RETURNS TEXT AS $$
DECLARE
    mac_address TEXT;
BEGIN
    mac_address := (
        SELECT string_agg(lpad(to_hex((random() * 255)::int), 2, '0'), ':')
        FROM generate_series(1, 6)
    );
    RETURN mac_address;
END;
$$ LANGUAGE plpgsql;

update equipo set mac = generate_random_mac();
